./echo.elf
  -> execute l'interp en mode REPL

./echo.elf -
./echo.elf srcfile
  -> execute l'interp

./echo.elf\ -i -
./echo.elf\ -i srcfile
  -> execute l'interp en mode interactif

---

./echo.elf --
  -> execute l'embed (echo.ml)

./echo.elf -- <arg(s)>
  -> execute l'embed

./echo.elf\ -i --
  -> execute l'embed en mode interactif

./echo.elf\ -i -- <arg(s)>
  -> execute l'embed en mode interactif

---

./interp.elf -- <arg(s)>
  -> fallback sur le mode REPL (utile pour passer des arguments en mode REPL)
